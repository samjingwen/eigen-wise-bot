\noindent
\begin{minipage}{0.65\linewidth}
Suppose $N \in \mathcal{L}(V)$ is nilpotent and there are vectors $v_1,\ldots,v_n \in V$ and nonnegative integers $m_1,\ldots,m_n$ such that
$N^{m_1}v_1,\ldots,Nv_1,v_1,\ldots,N^{m_n}v_n,\ldots,Nv_n,v_n$ is a basis for $V$ and
$N^{m_1+1}v_1=\cdots=N^{m_n+1}v_n=0$. Which of the following is necessarily equal to $n$?
\end{minipage}

\begin{enumerate}
  \item $\max\{\dim(\mathrm{null}(N^{m_i})) : 1 \le i \le n\}$
  \item $\dim(\mathrm{null}(N))$
  \item $\dim(\mathrm{range}(T))$
  \item $\dim(V)$
\end{enumerate}
